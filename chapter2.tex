\chapter{Mineral Resources (First Floor)}

\section{Concepts of Mineral Resources}

\subsection{Energy minerals}
Crude oil and coal of all types.

\subsection{Metallic minerals}
Iron, titanium, aluminum, copper, lead, zinc, tin, wolfram, molybdenum, gold, rare metals, etc.

\subsection{Non-metallic minerals}
Kaolin, glass sand, silica sand, asbestos, apatite, pyrite, phosphorite, construction materials, etc.

\subsection{Precious and semi-precious stones}
Ruby, sapphire, topaz, quartz, etc.

\subsection{Mineral water}
Mineral springs and water sources.
