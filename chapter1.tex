\chapter{General Geology (Ground Floor)}

\section{Geological Evolution According to the Old Geochronological Scale (Not Yet Updated on the Exhibit Panels)}

\subsection{Azoic Stage}
\begin{itemize}
    \item Anoxic Stage: from 4 billion years to 570 million years ago ($\sim$600 million years)
    \item Archeozoic Eon: from 4 billion years to 2.5 billion years ago ($\sim$2.6 billion years)
    \item Proterozoic Eon: from 2.5 billion years to 570 million years ago
\end{itemize}
\subsection{Phanerozoic Eon: from 570 million years ago to present}
\begin{itemize}
    \item Paleozoic Era: 570 million years to 250 million years ago
    \item Mesozoic Era: 250 million years to 64 million years ago
    \item Cenozoic Era: 64 million years ago to present
\end{itemize}

\section{Geological Processes: Introduction to 6 Processes}

\subsection{Cosmic Processes}
Displaying meteorites.

\subsection{Tectonic Processes}
Showing tectonic maps, folded and faulted rock samples, illustrations of deformation and contact metamorphism.

\subsection{Magmatic Processes}
Displaying intrusive and extrusive igneous rocks, volcanic rocks, and intrusive magma bodies.

\subsection{Metamorphic Processes}
Displaying metamorphic rocks and diagrams describing contact metamorphism and hydrothermal processes.

\subsection{Sedimentary Processes}
Displaying sedimentary rocks, clastic rocks, chemical and biochemical sedimentary rocks.

\subsection{Weathering Processes}
Displaying samples of chemical and physical weathering.
